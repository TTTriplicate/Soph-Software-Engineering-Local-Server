CREATE DATABASE SOCALSTYLE;

USE SOCALSTYLE;

CREATE TABLE SEX(
   SEXID VARCHAR(20),
   PRIMARY KEY(SEXID)
);

CREATE TABLE TSHIRTTYPE(
   TYPEID VARCHAR(25),
   PRICE DECIMAL(6,2),
   PRIMARY KEY(TYPEID)
);

CREATE TABLE SIZES(
   SIZEID VARCHAR(25),
   PRIMARY KEY(SIZEID)
);

CREATE TABLE COLOR(
   COLORID VARCHAR(25),
   PRIMARY KEY(COLORID)
);

CREATE TABLE CUSTOMER(
   CUSTOMERID INT(20) NOT NULL AUTO_INCREMENT,
   FNAME VARCHAR(20),
   LNAME VARCHAR(20),
   PHONENUMBER VARCHAR(20),
   EMAILADRESS VARCHAR(40),
   PASSWORD VARCHAR(20),
   PRIMARY KEY(CUSTOMERID)
);

CREATE TABLE DESIGN(
   DESIGNID VARCHAR(20),
   DESCRIPTION VARCHAR(255),
   PRIMARY KEY(DESIGNID)
);

CREATE TABLE ITEM(
   ITEMID INT(20) NOT NULL AUTO_INCREMENT,
   COLORID VARCHAR(25),
   SIZEID VARCHAR(25),
   TYPEID VARCHAR(25),
   SEXID VARCHAR(20),
   DESIGNID VARCHAR(20),
   FILEPATH VARCHAR(255),
   PRIMARY KEY(ITEMID),
   FOREIGN KEY(COLORID) REFERENCES COLOR(COLORID),
   FOREIGN KEY(SIZEID) REFERENCES SIZES(SIZEID),
   FOREIGN KEY(TYPEID) REFERENCES TSHIRTTYPE(TYPEID),
   FOREIGN KEY(SEXID) REFERENCES SEX(SEXID),
   FOREIGN KEY(DESIGNID) REFERENCES DESIGN(DESIGNID)
);

CREATE TABLE QUANTITY(
   ITEMID INT(20) NOT NULL,
   QUANTITY INT(20) NOT NULL,
   FOREIGN KEY(ITEMID) REFERENCES ITEM(ITEMID)
);
   
CREATE TABLE TRANSACTIONS(
   TRANSACTIONID INT NOT NULL AUTO_INCREMENT,
   CUSTOMERID INT(20) NOT NULL,
   DATE VARCHAR(20) NOT NULL,
   ITEMID VARCHAR(25),
   QUANTITY INT(20),
   PRIMARY KEY(TRANSACTIONID),
   FOREIGN KEY(CUSTOMERID) REFERENCES CUSTOMER(CUSTOMERID)
);

CREATE TABLE REVIEW(
   ITEMID INT(20),
   CUSTOMERID INT(20),
   STARS INT(20),
   REVIEWTEXT VARCHAR(255),
   FOREIGN KEY(ITEMID) REFERENCES ITEM(ITEMID),
   FOREIGN KEY(CUSTOMERID) REFERENCES CUSTOMER(CUSTOMERID)
);

CREATE TABLE CART(
   CUSTOMERID INT(20),
   ITEMID INT(20),
   QUANTITY INT(20),
   FOREIGN KEY(CUSTOMERID) REFERENCES CUSTOMER(CUSTOMERID),
   FOREIGN KEY(ITEMID) REFERENCES ITEM(ITEMID)
);

CREATE TABLE PAYMENTINFO(
   CUSTOMERID INT(20),
   CREDITCARD INT(20),
   DATE VARCHAR(20),
   CVV INT(10),
   ADDRESS VARCHAR(50),
   CITY VARCHAR(20),
   STATE VARCHAR(20),
   COUNTRY VARCHAR(20),
   ZIP INT (20),
   FOREIGN KEY(CUSTOMERID) REFERENCES CUSTOMER(CUSTOMERID)
);
